啊哈算法-第3章 第4节

题目：输入一个n，输出1~n的全排列

```js
function main(n){
  //book用于记录哪些牌已经被使用了，box表示盒子
  let book=new Array(n+1),box=new Array(n),arr=[];
  function dfs(step){
    //结束条件
    if(step===n){//已经走到最后一个盒子后面（表示所有的盒子都已经放入了牌）
      console.log(box);//打印出一种排列
      arr.push(box);
      return;//必须return 否则递归无法结束（book[i]又置为0，还可以继续放入后续盒子）
    }
    //遍历手中的牌（随着放入和回收，手中的牌在不断的变化）
    for(var i=1;i<=n;i++){
      //如果手中有这张牌
      if(book[i]!==1){
        box[step] = i;//将这张牌放入第step个盒子中
        book[i] = 1;//1表示手中已经没有这张牌
        dfs(step+1);//走到下一个盒子前，再放入下一张牌(递归调用，深度优先，会一直执行到最后一个盒子放入了牌之后，再执行后续代码)
        book[i] = 0;//所有盒子都放入牌之后，往回倒序收牌（当手中的牌回收一部分时，已置空的盒子可放入已收回的其他牌，再次递归到最后一个盒子后面，多出另一种排列）
      }
    }
  }
  //从第一个盒子开始
  dfs(0)
  console.log(arr.length)
}
main(9)
```
