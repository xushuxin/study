思路：

+ 先dom查找到所有元素，用对象记录每个标签出现的次数
+ 把对象转为数组
+ 对数组进行从大到小排序(不使用sort方法)，取前三

```js
var obj =[...document.querySelectorAll('*')].map(item=>item.nodeName).reduce((pre,cur)=>{
  cur in pre？pre[cur]++ : pre[cur]=1;
  return prev;
},{});
```

```js
var arr =[];
for(var key in obj){
     arr.push({[key]:obj[key]})
}
```

```js
//快速排序
function quickSort(ary){
    var arr =ary.slice();
    if(arr.length <=1 ) return arr;
    var mid = arr.splice([Math.floor(arr.length/2)],1)[0];
    var left=[],right=[];
    for(var i=0;i <arr.length;i++){
        if(Object.values(arr[i])[0]>Object.values(mid)[0]){
            left.push(arr[i]);
        }else{
            right.push(arr[i]);
        }
    }
    return quickSort(left).concat(mid,quickSort(right))
}
```
