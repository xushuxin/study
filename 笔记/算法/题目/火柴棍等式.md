题目

+ 0~9需要的火柴根数：[6,2,5,5,4,5,6,3,7,6]
+ 满足的等式A+B = C
+ 加号和等号各需要2根火柴
+ 所有火柴必须全部用上
+ 如果A不等于B,则A+B=C与B+A=C视为不同等式

计算总共有多少种拼法

```js
//m根火柴，可以拼出多少种加号等式
//1只需要2根火柴，题目限制最多24根，减去加号和等号需要的4根，20根，最多拼出10个1，A、B、C都不能超过1111
function main(m) {
  m = m - 4; //减去加号和等号需要的4根,获得实际可以拼成数字的根数
  var arr = [];
  for (var A = 0; A < 1111;A++) {
    for(var B = 0;B < 1111;B++){
      let C = A + B;//计算A+B的结果
      if(getCount(A)+getCount(B)+getCount(C) === m){//如果三个数需要的火柴根数满足要求
        arr.push(`${A}+${B}=${C}`);
      }
    }
  }
  return arr
}

/**
 * 
 * @param {数字} n
 * @return {数字n需要多少火柴拼成} count 
 */
function getCount(n){
  var countArr = [6,2,5,5,4,5,6,3,7,6];//记录0~9需要的火柴根数
  var count = countArr[n%10];//计算最后一位需要的火柴根数
  //如果大于10，需要加上每一位上数字需要的火柴数
  while(Math.floor(n/10)>0){
    count+=countArr[Math.floor(n/10)]%10;//加上左边一位需要的火柴根数
    n = Math.floor(n/10);//去掉最后一位，继续判断左边是否还有数字
  }
  return count
}
console.log(main(18))
```
