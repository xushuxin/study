深拷贝(简易版)

```js
function isObj(obj){
  return obj !==null && typeof obj === 'object'; 
}
function toType(obj){
  return ({}).toString.call(obj).match(/([^\s]+)\]$/)[1];
}
function deepClone(obj){
  if(!isObj(obj)) return obj;
  if(toType(obj)=== 'ArrayBuffer') return obj.slice();//deepClone ArrayBuffer
  if(toType(obj)!== 'Object') return new obj.constructor(obj); 
  let res = Array.isArray(obj)?new Array:new Object;
  for(var key in obj){
    res[key] = isObj(obj)?deepClone(obj[key]):obj[key];
  }
  return res;
}
var obj = {a:1,b:{name:2},arr:[{c:{d:1}}],e:new RegExp('\\d'),f:new Date()};
console.log(deepClone(obj))
```
